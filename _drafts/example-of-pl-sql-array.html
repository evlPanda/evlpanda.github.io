declare



    type table_list is table of varchar(128) index by binary_integer;

    tabs table_list;

    i number;



begin



    tabs(1) := 'aaa';

    tabs(2) := 'bbb';

    tabs(3) := 'ccc';

    

    for i in 1..3

    loop

        dbms_output.put_line(tabs(i));

    end loop;



end;
