FizzBuzz is a simple exercise and a nice illustration of how to refactor.

Version 1


def fizzbuzz(fizz, buzz, r):

    for i in range(0, r+1):
        result= ''
        if i % fizz == 0:
            result = 'Fizz'
        if i % buzz == 0:
            result += 'Buzz'

    print(str(i) + ': ' + result)


fizzbuzz( 3, 5, 100 )

Version 2

And after the mandatory "What if I want to buzz on 7, and also wizz on 9, etc.?". The function is now refactored to take a dictionary of any number of fizz, buzz, wizz, whatevers to look for.


def fizzbuzz2(searchDict, r):

    for i in range(1, r+1):
        result= ''
        for j in range(0, len(searchDict)):
            if i % searchDict[j]['Number'] == 0:
            result += searchDict[j]['Reply']

        print(str(i) + ': ' + result)


arrDict = [
    {'Number': 3, 'Reply': 'Fizz'},
    {'Number': 5, 'Reply': 'Buzz'},
    {'Number': 9, 'Reply': 'Wizz'}]

fizzbuzz2(arrDict, 100 )
