INSERT INTO PS_SOME_TEST_TBL
SELECT EMPLID, 'Test ID' 
FROM
    (SELECT EMPLID FROM PS_JOB
    ORDER BY DBMS_RANDOM.VALUE)
WHERE ROWNUM < 101;
